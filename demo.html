4. Lista de Tareas (r-for)

No hay tareas todavía

Total: tareas

<div r-data="ListaTareas()">
    <input r-model="nuevaTarea" @keyup.enter="agregar()" placeholder="Agregar una tarea...">
    <button @click="agregar()">Agregar</button>
    <ul>
        <template r-for="(tarea, index) in tareas">
            <li>
                <input type="checkbox" :checked="tarea.completada" @change="alternar(index)">
                <span :style="{ textDecoration: tarea.completada ? 'line-through' : 'none' }"
                      r-text="tarea.texto"></span>
                <button @click="eliminar(index)">Eliminar</button>
            </li>
        </template>
    </ul>
    <p r-show="tareas.length === 0">No hay tareas todavía</p>
</div>

<script>
function ListaTareas() {
    return {
        tareas: [],
        nuevaTarea: '',
        agregar() {
            if (this.nuevaTarea.trim()) {
                this.tareas.push({ texto: this.nuevaTarea, completada: false });
                this.nuevaTarea = '';
            }
        },
        eliminar(index) { this.tareas.splice(index, 1); },
        alternar(index) { this.tareas[index].completada = !this.tareas[index].completada; }
    }
}
</script>
